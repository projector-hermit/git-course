# Список команд для терминала и их описание

ls –  просмотр содержимого активного каталога:
    ls -l –  в виде списка с описанием; 
    ls -a – показать  скрытые системой файлы.

cat – просмотр содержимое указанного файла:
    cat -n – c нумерацией строк.

cd – переход в указанную директорию.
pwd – определить текущую директорию.
mkdir – создать новую папку.
file – подробная информация о файле.
cp – сопировать папку или файл:
    cp -r – рекурсивное копирование;
    cp -a – сохрание атрибутов документа;
    cp -i – с запросом на замену.
mv – перемещение или переименование папки/файла.
rm – удаляет файл или папку:
    rm -rf – рекурсивно и принудительно.
ln – создать ссылку на файл.
chmod – изменить права.
chown – сменить владельца.
locate – поиск в updatedb.
find – поиск.
df – анализ дискового пространства.
date – выводит дату и время.
sudo, sudo su – запуск от имени пользователя/СуперПользователя.
sleep – переводит компьютер в спящий режим.
shutdown — выключение.
reboot — перезагрузка.
ping – диагностика подключения.
echo – вывод строки текста в терминал.
nano – запуск текстового редактора.
htop – список запущенных процессов.
